## Install

```bash
npm install --save abakus
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'abakus'
import 'abakus/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
